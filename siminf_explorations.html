SimInf: Stochastic Population Network

One Population Stochastic SEIR metapopulation model with SimInf

Specification of a SEIR model, which contains three compartments susceptible (S), infected but not infectious (E), infectious (I) and recovered (R). The transmission route of infection to susceptible individuals is through direct contact between susceptible and infected individuals. The SEIR model has three state transitions given by: \[ \begin{aligned} & S \rightarrow E: \frac{\beta S I}{S + E + I + R} \\ & E \rightarrow I: \sigma E \\ & I \rightarrow R: \gamma I \end{aligned} \] where \(\beta\) is the transmission rate, \(\sigma\) is the incubation rate and \(\gamma\) is the recovery rate.

siminf_pop1 <- function(N = 1e6, I0 = 1, v = 0,
                        beta = 5, sigma = 1/7, gamma = 1/7, tf = 100, sims = 1) {
  vaccinated <- round(v * (N - I0))
  u0 <- data.frame(S = N - I0 - vaccinated, E = 0, I = I0, R = vaccinated)
  tspan <- seq(0, tf, by = 1)
  model <- SEIR(u0 = u0, tspan = tspan, beta = beta, epsilon = sigma, gamma = gamma)
  
  simulation <- function(x) {
    result <- run(model = model); return(trajectory(model = result))
  }
  purrr::rerun(sims, simulation())
}

We can see from the single simulation that the SEIR models implemented by SimInf and adaptivetau provide similar results. Given that the two implementations of the model provide similar results, we would like to know the relative run times of each of the models.

The following function runs any expression expr n times in parallel on mc.cores:

mcreplicate <- function(n, expr, mc.cores = NULL) {
  if (is.null(mc.cores)) mc.cores <- parallel::detectCores() - 1
  parallel::mclapply(integer(n), eval.parent(substitute(function(...) expr)), mc.cores = mc.cores)
}

The run-time to generate n realizations of the default SEIR model with parametes \(N = 1e6\), \(I = 0\), \(v = 0\), \(\beta = 5\), \(\sigma = 1/7\) and \(\gamma = 1/7\) over a 100 day time frame.

The following functions are used in the run-time routines.

# Returns the average time of peak infection in sims simulations of the SimInf SEIR model
f1 <- function(sims) {
  simulations <- siminf_pop1(sims = sims)
  mean(unlist(purrr::map(simulations, function(x) x[which.max(x$I), 2])))
}

# Returns the average time of peak infection in sims simulations of the adaptive SEIR model
f2 <- function(sims, N = 1e6, I0 = 1, v = 0, beta = 5, sigma = 1/7, gamma = 1/7, tf = 100, exact = FALSE) {
  simulations <- purrr::rerun(n, epiNetwork::seir(N = N, I0 = I0, v = v, beta = beta, 
                                                 sigma = sigma, gamma = gamma, tf = tf))
  mean(unlist(purrr::map(simulations, function(x) x[which.max(x$I), 1])))
}

Two Population Stochastic SEIR metapopulation model with SimInf

Next, we will consider a two subpopulation stochastic SEIR metapopulation model with mparse in SimInf.

siminf_pop2 <- function(N = c(1e6, 1e6), I0 = c(0,1), v = c(0.95, 0.5),
                        beta = 5, sigma = 1/7, gamma = 1/7, 
                        coupling = epiNetwork::make_coupling(n = 2, N = c(1e6, 1e6), 
                                                             coeffs = matrix(c(0.998, 0.002, 0.1, 0.9), 
                                                                             nrow = 2, ncol = 2)), 
                        tf = 100, sims = 1) {
  
  # All state transitions
  transitions <- c("S1 -> beta * (c11 * S1 * I1 / N1 + c12 * S1 * I2 / N2) -> E1", 
                   "E1 -> sigma * E1 -> I1", 
                   "I1 -> gamma * I1 -> R1", 
                   "S2 -> beta * (c21 * S2 * I1 / N1 + c22 * S2 * I2 / N2) -> E2", 
                   "E2 -> sigma * E2 -> I2",
                   "I2 -> gamma * I2 -> R2")
  # Compartments 
  compartments <- c("S1", "E1", "I1", "R1", "S2", "E2", "I2", "R2")
  
  # Initial condition vector
  initialization <- function(N, I0, v) {
    vaccinated <- unlist(purrr::map2(v, N, function(x, y) round(x * (y - 1))))
    x0 <- unlist(purrr::pmap(list(I0, vaccinated, N), function(x, y, z) c(z - x - y, 0, x, y)))
    names(x0) <- unlist(purrr::map(seq(1, length(N)), function(x) c(paste0("S", x), paste0("E", x),
                                                                    paste0("I", x), paste0("R", x))))
    data.frame(as.list(x0))
  }
  u0 <- initialization(N = N, I0 = I0, v = v)
  
  # Global data
  gdata = c(beta = beta, sigma = sigma, gamma = gamma, c11 = coupling[1, 1], 
            c21 = coupling[2, 1], c12 = coupling[1, 2], c22 = coupling[2, 2], 
            N1 = N[1], N2 = N[2])
  
  # Construct the model
  model <- mparse(transitions = transitions, compartments = compartments, 
                  gdata = gdata, u0 = u0, tspan = 1:tf)
  
  # Run the simulations and return a list of the trajectories
  simulation <- function(x) {
    result <- run(model = model); return(trajectory(model = result))
  }
  purrr::rerun(sims, simulation())
}

Lets calculate the average time until the peak of the epidemic in population 1 and population 2 in siminf model and the epiNetwork model. This will also serve as an opportunity to compare the running times of each of the models.

# Returns the average time of peak infection in sims simulations of the SimInf SEIR model
f1 <- function(sims) {
  simulations <- siminf_pop2(sims = sims)
  colMeans(do.call(rbind, purrr::map(simulations, function(x) c(x[which.max(x$I1), 2],x[which.max(x$I2), 2]))))
}

# Returns the average time of peak infection in sims simulations of the adaptivetau SEIR model
f2 <- function(sims, n = 2, N = c(1e6, 1e6), I0 = c(0,1), v = c(0.95, 0.5), 
               beta = 5, sigma = 1/7, gamma = 1/7, coupling, tf = 100, exact = FALSE) {
  simulations <- purrr::rerun(sims, epiNetwork::seir_popN(n = n, N = N, I0 = I0, 
                                                          v = v, beta = beta, sigma = sigma, 
                                                          gamma = gamma, coupling = coeffs, tf = tf))
  colMeans(do.call(rbind, purrr::map(simulations, function(x) c(x[which.max(x$I1), 1],x[which.max(x$I2), 1]))))
}
